2) initialize dvc
    - dvc init
    - dvc config core.analytics false # Disable analytics
    - git commit -m "DVC Initialization"
    - mkdir dvc
    - mkdir dvc/date
    - mkdir dvc/models
    - dvc add ./dvc
    - dvc config core.autostage true #to activate dvc autostaging
    - dvc remote add -d <storage_name> <protocol>://<user>@<server_name>:<port>/<directory_path>
